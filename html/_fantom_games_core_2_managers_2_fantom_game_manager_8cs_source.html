98 private bool ApplySettings(FantomGameSettings settings, out ReturnCode<bool> applyResult)
99 {
100 Log("GM) Trying to apply new settings.");
101
102 // basic check the settings can do on their own
103 // avoid loading if clearly won't be good
104 var setValid = settings.CheckValidity();
105 if (!setValid)
106 {
107 Log($"GM) Settings are not valid: {setValid.Message}.");
108
109 StateMachine.ChangeState(GameStates.Fail);
110 applyResult = setValid;
111 return false;
112 }
113
114 // other validity checks (TODO)
115 if (BoardManager == null && settings.GameBoardLoader == null)
116 {
117 Log($"GM) Does not have a valid Playing Game Board.");
118
119 StateMachine.ChangeState(GameStates.Fail);
120 applyResult = new(false, ReturnCodes.BadArgument,
121 $"No playing board is loaded and no board loader given in settings.");
122 return false;
123 }
124
125 // TODO: catch error loading the board (bad indices etc.)
126 if (settings.GameBoardLoader != null)
127 {
128 Log($"GM) Setting a new Game Board.");
129
130 BoardManager = new FantomBoard(settings.GameBoardLoader);
131 }
132
133 // Board at this point cannot be null
134 Debug.Assert(BoardManager != null);
135
136 Log($"GM) Resetting Round Context Manager.");
137 if (RoundContextManager == null)
138 {
139 RoundContextManager = new(
140 settings.SeekersCount,
141 settings.DetectivesCount,
142 BoardManager)
143 {
144 ActorsManager = ActorsManager,
145 TicketsManager = TicketsManager,
146 };
147 }
148 else
149 {
150 RoundContextManager.Restart(
151 settings.SeekersCount,
152 settings.DetectivesCount,
153 BoardManager
154 );
155 }
156
157 Log($"GM) Resetting Actor Manager.");
158
159 ActorsManager.Reset(1 + settings.SeekersCount, BoardManager.TileCount);
160
161 // change to a new logger if given
162 if (settings.LoggerDestination != null)
163 _activeLogger = settings.LoggerDestination;
164
165
166
167 Log($"GM) Settings applied.");
168
169 // if valid
170 ActiveSettings = settings;
171
172 // Ok
173 applyResult = new(true);
174 return true;
175 }

185 private void PrepareGame()
186 {
187 Log("GM) Preparing game.");
188
189 _stateId = Random.Shared.NextInt64();
190
191 // These shouldn't be null if Prepare is ever called
192 Debug.Assert(BoardManager != null);
193
194 GameContext = new(BoardManager)
195 {
196 ActorsManager = ActorsManager,
197 TicketsManager = TicketsManager,
198 GameManager = this,
199
200 KeepOrder = ActiveSettings.SeekerOrder,
201 SeekersCount = ActiveSettings.DetectivesCount + ActiveSettings.BobbiesCount,
202 DetectivesCount = ActiveSettings.DetectivesCount,
203 MaxMoves = ActiveSettings.MaxMoves,
204 MaxRounds = ActiveSettings.MaxRounds,
205 RevealingMoves = ActiveSettings.RevealingMoves
206 };
207
208 Debug.Assert(RoundContextManager != null);
209 RoundContextManager.Reset();
210
211 Log("GM) Setting Tickets.");
212 TicketsManager.Reset(1 + ActiveSettings.DetectivesCount);
213
214 Log("GM) Setting Supply Tickets.");
215 // set supply tickets
216 TicketsManager.SetSupplyTickets(ActiveSettings.SupplyTickets);
217
218 Log("GM) Setting Fantom Tickets.");
219 // give Fantom his starting Tickets
220 TicketsManager.SetActorTickets(0, ActiveSettings.FantomStartingTickets);
221
222 Log("GM) Setting Detectives Tickets.");
223 // give Detectives their Tickets
224 for (int i = 0; i < ActiveSettings.DetectivesCount; i++)
225 TicketsManager.SetActorTickets(1 + i, ActiveSettings.DetectiveStartingTickets);
226
227 Log("GM) Replacing Actors.");
228 // reset Actor positions
229 ActorsManager.ResetAllActors();
230
231 // place Actors if positions are given
232 if (ActiveSettings.FantomStartingPositions != null)
233 {
234 // Move Fantom to one of the random positions.
235 var pos = ActiveSettings.FantomStartingPositions[Random.Shared.Next(ActiveSettings.FantomStartingPositions.Length)];
236
237 Log($"GM) Placing Fantom at {pos}.");
238 ActorsManager.Move(0, pos);
239 }
240
241 if (ActiveSettings.SeekerStartingPositions != null)
242 {
243 // From settings we have enough to cover all detectives and bobbies
244 List<int> possiblePositions = [.. ActiveSettings.SeekerStartingPositions];
245
246 for (
247 int seekerIndex = 0;
248 seekerIndex < ActiveSettings.DetectivesCount + ActiveSettings.BobbiesCount;
249 ++seekerIndex)
250 {
251 // possibly slow, but mutual exclusion guaranteed
252 int randomIndex = Random.Shared.Next(possiblePositions.Count);
253 int pos = possiblePositions[randomIndex];
254
255 Log($"GM) Placing Seeker {seekerIndex} at {pos}.");
256
257 ActorsManager.Move(1 + seekerIndex, pos);
258 possiblePositions.RemoveAt(randomIndex);
259 }
260 }
261
262 Log("GM) Game prepared.");
263 StateMachine.ChangeState(GameStates.Choosing);
264 }

294 private ReturnCode<FantomGameState> GetGameState(bool isPublic)
295 {
296 var c_move = CurrentMove();
297 var c_round = CurrentRound();
298
299 int[] seeker_positions = new int[ActiveSettings.DetectivesCount + ActiveSettings.BobbiesCount];
300 int? fantom_position = null;
301 int[,] actor_tickets = new int[1 + ActiveSettings.DetectivesCount, FantomGameSettings.TICKET_KINDS_COUNT];
302 int[] supply_tickets = new int[FantomGameSettings.TICKET_KINDS_COUNT];
303
304 // Always fine
305 Debug.Assert(GameContext != null);
306
307 // if not valid, wouldn't make any sense to fill out, so just leave empty
308 if (IsValid())
309 {
310 // Fantom position if private or is revealing
311 if (!isPublic || GameContext.IsRevealing())
312 fantom_position = ActorsManager.GetTileOf(0);
313
314 // fill out Seekers' positions
315 for (int i = 0; i < seeker_positions.Length; ++i)
316 seeker_positions[i] = ActorsManager.GetTileOf(1 + i);
317
318 // actor Tickets
319 for (int i = 0; i < 1 + ActiveSettings.DetectivesCount; ++i)
320 for (int c = 0; c < FantomGameSettings.TICKET_KINDS_COUNT; ++c)
321 actor_tickets[i, c] = TicketsManager.GetActorTickets(i, (TicketKinds)c);
322
323 // supply Tickets
324 for (int c = 0; c < FantomGameSettings.TICKET_KINDS_COUNT; ++c)
325 supply_tickets[c] = TicketsManager.GetSupplyTickets((TicketKinds)c);
326 }
327
328 var gameState = new FantomGameState(
329 ActiveSettings.DetectivesCount,
330 ActiveSettings.BobbiesCount,
331 seeker_positions,
332 fantom_position,
333 GameContext.LastPublicFantomPosition,
334 actor_tickets,
335 supply_tickets
336 )
337 {
338 CurrentMove = c_move,
339 CurrentRound = c_round
340 };
341
342 return IsValid() ? new(gameState) : new(gameState, ReturnCodes.InvalidState);
343

399 private void CheckGameOver()
400 {
401 // The only time a game over check happens is if we are in a valid state, so never will be null
402 Debug.Assert(GameContext != null);
403
404 if (GameContext.CheckConditions(ActiveSettings.SeekersVictoryConditions))
405 {
406 GameContext.SeekersWon(GameContext.GetPassingConditions(ActiveSettings.SeekersVictoryConditions));
407 Log("GM) Seekers have won the game.");
408 }
409 else if (GameContext.CheckConditions(ActiveSettings.FantomVictoryConditions))
410 {
411 GameContext.FantomWon(GameContext.GetPassingConditions(ActiveSettings.FantomVictoryConditions));
412 Log("GM) Fantom has won the game.");
413 }
414 else if (GameContext.CheckConditions(ActiveSettings.DrawConditions))
415 {
416 GameContext.Draw(GameContext.GetPassingConditions(ActiveSettings.DrawConditions));
417 Log("GM) The game is a draw.");
418 }
419
420 // Avoid firing the event multiple times
421 if (StateMachine.CurrentState != GameStates.GameOver &&
422 GameContext.gameResolutionResult != GameResolutionStates.Playing)
423 {
424 Log("GM) Game is over.");
425
426 StateMachine.ChangeState(GameStates.GameOver);
427 }

832 ReturnCode<bool> IFantomPlayerInterface.Move(int tileIndex_Where, TicketKinds ticketKind)
833 {
834 Log($"FI) Fantom Move request to {tileIndex_Where} via {Enum.GetName(ticketKind)}.");
835
836 lock (_lock)
837 {
838 Log($"FI) Fantom Move request to {tileIndex_Where} via {Enum.GetName(ticketKind)} processing.");
839
840 // Fantom can only move in Fantom Turn, Double1, Double2 states
841 // Fantom can only move to free tiles, else the Game would be over
842 // Can only move to tiles, where there's a path between the current position and the wanted one
843 // Has to have at least one Ticket of that kind
844
845 // Certain Tickets currently unsupported or have to be used in a different way
846
847 if (RoundContextManager == null)
848 return new(false, ReturnCodes.InvalidState,
849 "The game isn't properly initialized yet.");
850
851 if (StateMachine.CurrentState != GameStates.FantomTurn &&
852 StateMachine.CurrentState != GameStates.FantomDouble1 &&
853 StateMachine.CurrentState != GameStates.FantomDouble2)
854 return new(false, ReturnCodes.InvalidState,
855 $"Fantom cannot move in the current state {StateMachine.CurrentState}.");
856
857 if (!RoundContextManager.FantomMove(tileIndex_Where, ticketKind, out ReturnCode<bool> moveResult))
858 {
859 Log($"FI) Fantom Move request denied: {moveResult.Message}");
860 return moveResult;
861 }
862
863 Log($"FI) Fantom Moved to {tileIndex_Where} via {Enum.GetName(ticketKind)}.");
864
865 // If used a Double Ticket or not:
866 // Based on the current state, transition to the next
867 if (StateMachine.CurrentState == GameStates.FantomDouble1)
868 StateMachine.ChangeState(GameStates.FantomDouble2);
869 else
870 StateMachine.ChangeState(GameStates.PostFantomTurn);
871
872 // Ok
873 return new(true);
874 }
875 }

877 ReturnCode<int> IFantomPlayerInterface.CannotMove()
878 {
879 Log("FI) Fantom No Move request.");
880
881 lock (_lock)
882 {
883 Log("FI) Fantom No Move request processing.");
884
885 if (GameContext == null)
886 return new(-1, ReturnCodes.InvalidState,
887 "The game isn't properly initialized yet.");
888
889 // Fantom cannot move if he has no Tickets or is Blocked by the Seekers
890 // This might lead to a game over, but only when in a state, where this is checked
891 // To get to it, a confirmation from the Fantom is required - I cannot move
892
893 // Can Move -> this operation is not allowed
894

944 ReturnCode<bool> ISeekersPlayerInterface.Move(int seekerIndex, int tileIndex_Where, TicketKinds ticketKind)
945 {
946 Log($"SI) Seeker {seekerIndex} Move request to {tileIndex_Where} via {Enum.GetName(ticketKind)}.");
947
948
949 lock (_lock)
950 {
951 Log($"SI) Seeker {seekerIndex} Move request to {tileIndex_Where} via {Enum.GetName(ticketKind)} processing.");
952
953 // Can only use the Mode1, Mode2, Mode3
954
955 // Can only move in DetectiveTurn state
956 // If the ordering of Detectives is enabled, can only move the current one
957 // A Seeker can only move if it hasn't moved yet this round
958 // Detectives can only move if they have enough tickets
959
960 // Seekers can only move to tiles that are free, or have the Fantom on them
961 // Seekers can move to tiles that are connected by the specified mode
962
963 // After moving, Detectives give their Tickets to the Fantom
964
965 if (RoundContextManager == null || GameContext == null)
966 return new(false, ReturnCodes.InvalidState,
967 "The game isn't properly initialized yet.");
968
969 if (StateMachine.CurrentState != GameStates.SeekersTurn)
970 return new(false, ReturnCodes.InvalidState,
971 $"Seekers cannot move in the current state {StateMachine.CurrentState}.");
972
973 if (!GameContext.IsValidSeekerIndex(seekerIndex))
974 return new(false, ReturnCodes.BadArgument,
975 $"Seeker {seekerIndex} is not a valid seeker index.");
976
977 if (!GameContext.IsThisSeekersTurn(seekerIndex))
978 return new(false, ReturnCodes.InvalidState,
979 $"Seeker {seekerIndex} cannot play as it is {GameContext.CurrentSeekerIndex}'s turn.");
980
981 // Try to do the Move, if it fails, inform caller
982 if (!RoundContextManager.SeekerMove(seekerIndex, tileIndex_Where, ticketKind,
983 out ReturnCode<bool> moveResult))
984 {
985 Log($"SI) Seeker {seekerIndex} Move request denied: {moveResult.Message}");
986 return moveResult;
987 }
988
989 Log($"SI) Seeker {seekerIndex} Moved to {tileIndex_Where} via {Enum.GetName(ticketKind)}.");
990
991 // Update state
992 StateMachine.ChangeState(GameStates.PostSeekerTurn);
993
994 // Ok
995 return new(true);
996 }

999 ReturnCode<int> ISeekersPlayerInterface.CannotMove()
1000 {
1001 Log("SI) Seeker No Move request.");
1002
1003 lock (_lock)
1004 {
1005 Log("SI) Seeker No Move request processing.");
1006
1007 if (RoundContextManager == null || GameContext == null)
1008 return new(-1, ReturnCodes.InvalidState,
1009 "The game isn't properly initialized yet.");
1010
1011 // If ordering matters and the next Seeker cannot Move, then this method can be used
1012 // If ordering doesn't matter and
1013 // If All unmoved Seekers Cannot Move then the round can be ended via this method
1014 // If at least one that can, then this is invalid
1015 // If no Seekers can Move, this should be a game over condition, but that is checked only
1016 // in certain states, so to get to them, this method can be used
1017
1018 ReturnCode<int> canResult;
1019
1020 if (ActiveSettings.SeekerOrder)
1021 {
1022 Log("SI) Seeker play order enabled so current seeker is checked only.");
1023
1024 // The one that is currently to play
1025 if (!RoundContextManager.HasMoved(GameContext.CurrentSeekerIndex) &&
1026 GameContext.CanActorMove(1 + GameContext.CurrentSeekerIndex, out canResult))
1027 {
1028 Log($"SI) Seeker No Move request denied - Seeker {GameContext.CurrentSeekerIndex} can Move to {canResult.Value}.");
1029 return new(canResult.Value, ReturnCodes.Fail,
1030 $"Seeker {GameContext.CurrentSeekerIndex} can Move to {canResult.Value} so No Move isn't possible.");
1031 }
1032
1033 Log($"SI) Seeker {GameContext.CurrentSeekerIndex} cannot Move, so skipping turn.");
1034
1035 // Mark this one as having Moved
1036 RoundContextManager.Moved(GameContext.CurrentSeekerIndex);
1037
1038 }
1039 else
1040 {
1041 Log("SI) Seeker play order disabled so current all seekers have to be checked.");
1042
1043 for (int i = 0; i < RoundContextManager.SeekersCount; ++i)
1044 {
1045 // One that hasn't moved, can move
1046 if (!RoundContextManager.HasMoved(i) &&
1047 GameContext.CanActorMove(1 + i, out canResult))
1048 {
1049 Log($"SI) Seeker No Move request denied - Seeker {i} can Move to {canResult.Value}.");
1050 return new(canResult.Value, ReturnCodes.Fail,
1051 $"Seeker {i} can Move to {canResult.Value} so No Move isn't possible.");
1052 }
1053 }
1054
1055 Log("SI) All Seekers cannot Move, skipping now.");
1056
1057 // Mark all as having Moved
1058 for (int i = 0; i < RoundContextManager.SeekersCount; ++i)
1059 RoundContextManager.Moved(i);
1060 }
1061
1062 StateMachine.ChangeState(GameStates.PostSeekerTurn);
1063
1064 // Ok
1065 return new(-1);
1066 }
1067 }
