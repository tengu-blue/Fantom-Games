41 private FantomIntermediary(
42 IFantomGamesFacade livingPlayerFacade,
43 IntermediarySettings iSettings,
44 FantomGameSettings fSettings)
45 {
46 _activeSettings = iSettings;
47
48 // based on which role the Living Player is playing, set the visibility of the passed Facade
49 _managedFacades.Add(new() { GameFacade = livingPlayerFacade, IsPublic = !iSettings.LivingPlayerIsFantom });
50
51 // initial game setup
52 var gm = FantomGameManager.CreateGame(fSettings);
53 if (gm.Code != ReturnCodes.OK)
54 SendError(gm.Message);
55 _gameManager = gm.Value;
56
57 _gameManager.AddGameStateChangeCallback(StateChanged);
58
59 var gameBoard = _gameManager.GetBoard();
60 Debug.Assert(gameBoard != null);
61
62 RecreateOpponents();
63 Debug.Assert(seeker != null && fantom != null);
64
65 _gameManager.Start();
66
67 if (iSettings.ComputerSeeker)
68 seeker.Wake();
69
70 if (iSettings.ComputerFantom)
71 fantom.Wake();
72 }

74 private void RecreateOpponents()
75 {
76 lock (_safetyLock)
77 {
78 var gameBoard = _gameManager.GetBoard();
79 Debug.Assert(gameBoard != null);
80
81 seeker?.Terminate();
82 fantom?.Terminate();
83
84 seeker = new SeekerOpponent(this, gameBoard, _gameManager.GetActiveSettings());
85 fantom = new FantomOpponent(this, gameBoard, _gameManager.GetActiveSettings());
86
87 // the 0-th one is always the user
88 if (_managedFacades.Count > 1)
89 {
90 _managedFacades.RemoveAt(2);
91 _managedFacades.RemoveAt(1);
92 }
93 _managedFacades.Add(new() { GameFacade = seeker, IsPublic = true });
94 _managedFacades.Add(new() { GameFacade = fantom, IsPublic = false });
95 }

166 private void StateChanged(long id, GameStates from, GameStates to)
167 {
168 if (from == GameStates.Choosing)
169 {
170 // Notify.
171 var gameStatePublic = GetPublicGameState();
172 var gameStatePrivate = GetPrivateGameState();
173
174 // in choosing, game state has to be fine
175 Debug.Assert(gameStatePublic != null);
176 Debug.Assert(gameStatePrivate != null);
177
178 _managedFacades.ForEachPublicOnly(f => f.GameStarted(gameStatePublic.Value));
179 _managedFacades.ForEachPrivate(f => f.GameStarted(gameStatePrivate.Value));
180
181 }
182
183 if (to == GameStates.GameOver)
184 {
185 // NOTE: if in GameOver state, result should be fine, but just in case...
186 var gameResult = _gameManager.GetGameResult();
187
188 if (gameResult.Code == ReturnCodes.OK)
189 _managedFacades.ForEachPublic(f => f.GameFinished(gameResult));
190 else
191 SendError(gameResult.Message);
192 }
193
194 if (to == GameStates.FantomTurn)
195 {
196 _managedFacades.ForEachPublic(f => f.FantomTurnBegin(_gameManager.CurrentMove()));
197 }
198
199 if (to == GameStates.SeekersTurn && from == GameStates.PostFantomTurn)
200 {
201 _managedFacades.ForEachPublic(f => f.FantomTurnEnd());
202 }
203
204 if (to == GameStates.SeekersTurn && _gameManager.GetActiveSettings().SeekerOrder)
205 {
206 _managedFacades.ForEachPublic(f => f.SeekerTurnBegin(SeekerIndex()));
207 }
208
209 if (from == GameStates.SeekersTurn && _gameManager.GetActiveSettings().SeekerOrder)
210 {
211 _managedFacades.ForEachPublic(f => f.SeekerTurnEnd(SeekerIndex()));
212 }
213
214 if (to == GameStates.RoundOver)
215 {
216 _managedFacades.ForEachPublic(f => f.RoundOver(_gameManager.CurrentRound()));
217 }
218 }

263 {
264 Monitor.Enter(_safetyLock);
265
266 var retCode = _gameManager.Restart(fSettings);
267
268 if (retCode)

270 RecreateOpponents();

272 if (_activeSettings.ComputerSeeker)
273 seeker.Wake();

275 if (_activeSettings.ComputerFantom)
276 fantom.Wake();
277

279
280 _managedFacades.ForEachPublic(f => f.GameRestarted(fSettings));
281 }
282 else
283 {
284 Monitor.Exit(_safetyLock);
285
286 SendError(retCode.Message);
287 }
288
