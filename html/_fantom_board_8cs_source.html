52 // iterate over all the tiles

58 foreach (var neighbor in tile.Mode1Neighbors)

68 _river_tileNeighbors.Add(BoundedIndex(neighbor));
69
70 ++index;
71
72 _mode1_tileNeighborsIndices[index] = _mode1_tileNeighbors.Count;
73 _mode2_tileNeighborsIndices[index] = _mode2_tileNeighbors.Count;
74 _mode3_tileNeighborsIndices[index] = _mode3_tileNeighbors.Count;
75 _river_tileNeighborsIndices[index] = _river_tileNeighbors.Count;
76
77 // force stop in case of more tiles than necessary
78 if (index == TileCount)

80 }

82 // not enough tiles given -> error
83 if (index != TileCount)
