127 public ReturnCode<bool> CheckValidity()
128 {
129
130 // At least one opponent else the game is pointless
131 if (DetectivesCount < 0 || BobbiesCount < 0 || DetectivesCount + BobbiesCount == 0)
132 {
133 return new(false, ReturnCodes.InvalidSettings,
134 $"There has to be at least one opponent for the Fantom. Got : {DetectivesCount} Detectives and {BobbiesCount} Bobbies.");
135 }
136
137 if (DetectivesCount + BobbiesCount > MAXIMUM_SEEKERS_COUNT)
138 {
139 return new(false, ReturnCodes.InvalidSettings,
140 $"There can only be {MAXIMUM_SEEKERS_COUNT} opponents for the Fantom at most. Got : {DetectivesCount + BobbiesCount}");
141 }
142
143 // At least one play cycle else the game is pointless
144 if (MaxMoves <= 0 || MaxRounds <= 0)
145 {
146 return new(false, ReturnCodes.InvalidSettings,
147 $"Game has to have at least one Round and Move. Got : {MaxMoves} Moves {MaxRounds} Rounds");
148 }
149
150 // If specified a set of positions, have to have at least enough to place all Actors
151 if (SeekerStartingPositions != null && SeekerStartingPositions.Length < DetectivesCount + BobbiesCount)
152 {
153 return new(false, ReturnCodes.InvalidSettings,
154 $"Not enough Detective Starting Positions given!");
155 }
156 if (FantomStartingPositions != null && FantomStartingPositions.Length < 1)
157 {
158 return new(false, ReturnCodes.InvalidSettings,
159 $"Not enough Fantom Starting Positions given!");
160 }
161
162 var actorsCount = 1 + DetectivesCount + BobbiesCount;
163 // At least as many tiles as Actors
164 if (GameBoardLoader != null && GameBoardLoader.TileCount < MAXIMUM_SEEKERS_COUNT + 1)
165 {
166 return new(false, ReturnCodes.InvalidSettings,
167 $"The given board does not have enough tiles for all Actors. Got : {GameBoardLoader.TileCount} need {actorsCount}");
168 }
169
170 // all passed
171 return new(true);
172 }
