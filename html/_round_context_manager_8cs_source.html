145 public bool PlaceSeekerAt(int seekerIndex, int tileIndex, out ReturnCode<bool> placeResult)
146 {
147
148 // Check indices
149 if (!CheckSeekerIndex(seekerIndex, out ReturnCode<bool> seekerResult))
150 {
151 placeResult = seekerResult;
152 return false;
153 }
154
155 if (!CheckBoardTileIndex(tileIndex, out ReturnCode<bool> indexResult))
156 {
157 placeResult = indexResult;
158 return false;
159 }
160
161 // If position not free -> can't move
162 if (ActorsManager.IsActorAt(tileIndex))
163 {
164 placeResult = new(false, ReturnCodes.PositionOccupied,
165 $"Tile position {tileIndex} occupied by another Actor {ActorsManager.GetActorAt(tileIndex)}, cannot place Seeker {seekerIndex}.");
166 return false;
167 }
168
169
170 // Place the detective at that position only if it is free
171 ActorsManager.Move(1 + seekerIndex, tileIndex);
172
173 placeResult = new(true);
174 return true;
175 }

182 public bool PlaceFantomAt(int tileIndex, out ReturnCode<bool> placeResult)
183 {
184 // Check indices
185 if (!CheckBoardTileIndex(tileIndex, out ReturnCode<bool> tileResult))
186 {
187 placeResult = tileResult;
188 return false;
189 }
190 // Fantom can be placed only after all the Detectives
191 for (int i = 1; i < ActorsManager.ActorsCount; ++i)
192 if (!ActorsManager.IsActorOnBoard(i))
193 {
194 placeResult = new(false, ReturnCodes.InvalidOperation,
195 $"The Fantom cannot be placed until all Seekers have been. Seeker {i - 1} isn't placed yet.");
196 return false;
197 }
198
199
200 // If position not free -> can't move
201 if (ActorsManager.IsActorAt(tileIndex))
202 {
203 placeResult = new(false, ReturnCodes.PositionOccupied,
204 $"Tile position {tileIndex} occupied by another Actor {ActorsManager.GetActorAt(tileIndex)}, cannot place Fantom.");
205 return false;
206 }
207
208 // Place Fantom at that position only if it is free
209 ActorsManager.Move(0, tileIndex);

211 placeResult = new(true);

237 public bool FantomMove(int tileIndex_Where, TicketKinds ticketKind,
238 out ReturnCode<bool> moveResult)
239 {
240
241 if (ticketKind == TicketKinds.Double || ticketKind == TicketKinds.Balloon)
242 {
243 moveResult = new(false, ReturnCodes.Fail,
244 $"The given Ticket kind cannot be used currently.");
245 return false;
246 }
247
248 // Index checking
249 if (!CheckBoardTileIndex(tileIndex_Where, out ReturnCode<bool> indexResult))
250 {
251 moveResult = indexResult;
252 return false;
253 }
254

261

275 public bool SeekerMove(int seekerIndex, int tileIndex_Where, TicketKinds ticketKind,
276 out ReturnCode<bool> moveResult)
277 {
278 if (ticketKind != TicketKinds.Mode1 && ticketKind != TicketKinds.Mode2 && ticketKind != TicketKinds.Mode3)
279 {
280 moveResult = new(false, ReturnCodes.BadArgument,
281 $"Seekers can only use the basic 3 Ticket kinds, not {ticketKind}.");
282 return false;
283 }
284
285 if (HasMoved(seekerIndex))
286 {
287 moveResult = new(false, ReturnCodes.BadArgument,
288 $"Seeker {seekerIndex} has already moved this Round.");
289 return false;
290 }
291
292 // Check indices
293 if (!CheckSeekerIndex(seekerIndex, out ReturnCode<bool> seekerResult))
294 {
295 moveResult = seekerResult;
296 return false;
297 }
298
299 if (!CheckBoardTileIndex(tileIndex_Where, out ReturnCode<bool> indexResult))
300 {
301 moveResult = indexResult;
302 return false;
303 }
304
305
306 if (!IsMoveValid(1 + seekerIndex, tileIndex_Where, ticketKind, out ReturnCode<bool> validResult))
307 {
308 moveResult = validResult;
309 return false;
310 }
311
312 // Move to the spot
313 ActorsManager.Move(1 + seekerIndex, tileIndex_Where);
314 Moved(seekerIndex);
315
316
317 // Detective Ticket handling
318 if (seekerIndex < DetectivesCount)
319 {
320 // Remove own Ticket
321 TicketsManager.RemoveActorTickets(1 + seekerIndex, ticketKind, 1);
322 // Give Ticket to Fantom if Detective
323 TicketsManager.AddActorTickets(0, ticketKind, 1);
324 }
325
326 moveResult = new(true);
327 return true;
328 }

333 public bool IsMoveValid(int actorIndex, int tileIndex_Where, TicketKinds ticketKind,
334 out ReturnCode<bool> moveResult)
335 {
336
337 if (actorIndex == 0)
338 {
339 // Fantom can only move to a free tile
340 if (ActorsManager.IsActorAt(tileIndex_Where))
341 {
342 moveResult = new(false, ReturnCodes.BadArgument,
343 $"Cannot move Fantom to a position {tileIndex_Where} with a Seeker on it.");
344 return false;
345 }
346 }
347 else
348 {
349 // Seekers can move to free tiles or the tile with Fantom
350 if (ActorsManager.IsActorAt(tileIndex_Where) && ActorsManager.GetActorAt(tileIndex_Where) != 0)
351 {
352 moveResult = new(false, ReturnCodes.BadArgument,
353 $"Cannot move Seeker {actorIndex - 1} to a position {tileIndex_Where} with another Seeker {ActorsManager.GetActorAt(tileIndex_Where) - 1} on it.");
354 return false;
355 }
356 }
357
358 // If not Bobby, has to have that Ticket available
359 if (actorIndex - 1 < DetectivesCount)
360 {
361 if (!ActorHasEnoughTickets(actorIndex, ticketKind, 1, out ReturnCode<bool> hasResult))
362 {
363 moveResult = hasResult;
364 return false;
365 }
366 }
367
368 // Always have to have a connection via the given Kind
369 if (ticketKind == TicketKinds.Black)
370 {
371 // Black Ticket -> Can use any transport
372 if (!BoardManager.IsNeighbor(ActorsManager.GetTileOf(actorIndex), tileIndex_Where))
373 {
374 moveResult = new(false, ReturnCodes.BoardTilesNotConnected,
375 $"Black Ticket cannot be used as tiles {ActorsManager.GetTileOf(actorIndex)} and {tileIndex_Where} are not connected.");
376 return false;
377 }
378 }
379 else
380 {
381 // Mode Tickets -> Can use that transport
382 if (!BoardManager.IsNeighbor(TicketKindToTravelMode(ticketKind), ActorsManager.GetTileOf(actorIndex), tileIndex_Where))
383 {
384 moveResult = new(false, ReturnCodes.BoardTilesNotConnected,
385 $"Ticket {ticketKind} cannot be used as tiles {ActorsManager.GetTileOf(actorIndex)} and {tileIndex_Where} are not connected via this Mode.");
386 return false;
387 }
388 }
389
390 moveResult = new(true);
391 return true;
