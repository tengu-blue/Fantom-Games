191 this.FantomLastKnownPosition = FantomLastKnownPosition ?? -1;
192
193 // Set Fantom Tickets
194 for (int i = 0; i < FantomGameSettings.TICKET_KINDS_COUNT; ++i)
195 {
196 FantomTickets[i] = ActorTickets[0, i];
197 this.SupplyTickets[i] = SupplyTickets[i];
198 }
